import hypothesis.strategies as st
from hypothesis import given, settings, assume

# Leaf strategies
def make_x_node():
    from primel.tree import Node
    return Node("x", lambda X: X, 0)

def make_const_node(v):
    from primel.tree import Node
    return Node("const", v, 0)

leaves = st.one_of([
    st.builds(lambda: make_x_node()),
    st.builds(lambda v: make_const_node(v), st.floats(min_value=-10, max_value=10, allow_nan=False, allow_infinity=False)),
])

# Unary operators
unary_ops = st.sampled_from([
    ("log", np.log),
    ("sqrt", np.sqrt),
    ("exp", np.exp),
])

# Binary operators
binary_ops = st.sampled_from([
    ("add", np.add),
    ("sub", np.subtract),
    ("mul", np.multiply),
    ("div", np.divide),
])

@st.composite
def expression_trees(draw, max_leaves=15):
    from primel.tree import Node, ExpressionTree
    
    def extend(base):
        return st.one_of([
            st.builds(
                lambda op, child: [Node(op[0], op[1], 1)] + child,
                unary_ops, base
            ),
            st.builds(
                lambda op, left, right: [Node(op[0], op[1], 2)] + left + right,
                binary_ops, base, base
            ),
        ])
    
    node_list = draw(st.recursive(
        st.builds(lambda n: [n], leaves),
        extend,
        max_leaves=max_leaves
    ))
    return ExpressionTree.init_from_list(node_list)
```

class TestLevelSetPreservation:
    """Property-based tests proving simplify preserves level set topology."""
    
    @given(tree=expression_trees(max_leaves=15))
    @settings(max_examples=200, deadline=None)
    def test_simplify_preserves_zeros(self, tree):
        """Core property: simplify(f)(x) == 0 ⟺ f(x) == 0"""
        X = np.linspace(-5, 5, 50).reshape(-1, 1)
        
        original_tree = tree.copy()
        with np.errstate(invalid='ignore', divide='ignore'):
            original = tree.evaluate(X)
        
        simplify_tree(tree, X)
        
        with np.errstate(invalid='ignore', divide='ignore'):
            simplified = tree.evaluate(X)
        
        original_zeros = np.isclose(original, 0, atol=1e-10) | np.isnan(original)
        simplified_zeros = np.isclose(simplified, 0, atol=1e-10) | np.isnan(simplified)
        
        np.testing.assert_array_equal(original_zeros, simplified_zeros,
            err_msg="Simplification changed zero locations")
    
    @given(tree=expression_trees(max_leaves=10))
    @settings(max_examples=100, deadline=None)
    def test_simplify_idempotent(self, tree):
        """Property: simplify(simplify(f)) == simplify(f)"""
        X = np.linspace(-5, 5, 30).reshape(-1, 1)
        
        simplify_tree(tree, X)
        once_simplified = tree.copy()
        
        simplify_tree(tree, X)
        twice_simplified = tree
        
        assert len(once_simplified.nodes) == len(twice_simplified.nodes)
        for n1, n2 in zip(once_simplified.nodes, twice_simplified.nodes):
            assert n1.name == n2.name
```

These tests verify the mathematical property that matters: zero locations are preserved.
